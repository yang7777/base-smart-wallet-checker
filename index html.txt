async function connectWallet() {

  if (!window.ethereum) {
    alert("No wallet detected");
    return;
  }

  try {

    const accounts = await window.ethereum.request({
      method: "eth_requestAccounts"
    });

    const address = accounts[0];

    const chainIdHex = await window.ethereum.request({
      method: "eth_chainId"
    });

    const chainId = parseInt(chainIdHex, 16);
const provider = new ethers.JsonRpcProvider(
  "https://mainnet.base.org"
);

const balanceWei =
  await provider.getBalance(address);

const balance =
  ethers.formatEther(balanceWei);

    let networkName = "Unknown";

    if(chainId === 8453){
      networkName = "Base Mainnet";
    }

    if(chainId === 1){
      networkName = "Ethereum";
    }

    if(chainId === 84532){
      networkName = "Base Sepolia";
    }

    // Smart Wallet Detection
    // 簡易判定

    let walletType = "EOA Wallet";

    if (
      window.ethereum.isCoinbaseWallet
    ) {
      walletType = "Coinbase Wallet";
    }

    // Smart Walletっぽい特徴を簡易表示
    // 実際には完全判定ではない

    const code = await window.ethereum.request({
      method: "eth_getCode",
      params: [address, "latest"]
    });

    if(code && code !== "0x"){
      walletType = "Smart Contract Wallet";
    }

    document.getElementById("address").innerText =
      address;

    document.getElementById("chainId").innerText =
      chainId;

    document.getElementById("network").innerText =
      networkName;

    const walletTypeElement =
      document.getElementById("walletType");

    walletTypeElement.innerText =
      walletType;

    if(walletType.includes("Smart")){
      walletTypeElement.className =
        "value success";
    } else {
      walletTypeElement.className =
        "value warning";
    }

  } catch(err) {

    console.error(err);
    alert("Connection failed");

  }

}

</script>
