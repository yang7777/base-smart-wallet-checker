async function connectWallet() {

  if (!window.ethereum) {
    alert("No wallet detected");
    return;
  }

  try {

    const accounts = await window.ethereum.request({
      method: "eth_requestAccounts"
    });

    const address = accounts[0];

let walletAge = "Unknown";

    const chainIdHex = await window.ethereum.request({
      method: "eth_chainId"
    });

    const chainId = parseInt(chainIdHex, 16);

const provider = new ethers.JsonRpcProvider(
  "https://mainnet.base.org"
);

try {

  const ensName =
    await provider.lookupAddress(address);

  if(ensName){
    basename = ensName;
  }

} catch(err) {

  console.log("No basename");

}

const balanceWei =
  await provider.getBalance(address);

const balance =
  ethers.formatEther(balanceWei);

const feeData =
  await provider.getFeeData();

const gasPrice =
  ethers.formatUnits(
    feeData.gasPrice,
    "gwei"
  );

const apiKey = "PXZMHFSXCDKWQTDTK5TYVHUISUV1MTBZKT";

const txUrl =
  `https://api.basescan.org/api
  ?module=account
  &action=txlist
  &address=${address}
  &startblock=0
  &endblock=99999999
  &page=1
  &offset=1
  &sort=asc
  &apikey=${apiKey}`;

const txResponse =
  await fetch(txUrl);

const txData =
  await txResponse.json();

if(
  txData.result &&
  txData.result.length > 0
){

  const firstTx =
    txData.result[0];

  const firstTimestamp =
    Number(firstTx.timeStamp) * 1000;

  const now =
    Date.now();

  const diffDays =
    Math.floor(
      (now - firstTimestamp)
      / (1000 * 60 * 60 * 24)
    );

  walletAge =
    diffDays + " days";
}

    let networkName = "Unknown";

    if(chainId === 8453){
      networkName = "Base Mainnet";
    }

    if(chainId === 1){
      networkName = "Ethereum";
    }

    if(chainId === 84532){
      networkName = "Base Sepolia";
    }

    // Smart Wallet Detection
    // 簡易判定

    let walletType = "EOA Wallet";

    if (
      window.ethereum.isCoinbaseWallet
    ) {
      walletType = "Coinbase Wallet";
    }

    // Smart Walletっぽい特徴を簡易表示
    // 実際には完全判定ではない

    const code = await window.ethereum.request({
      method: "eth_getCode",
      params: [address, "latest"]
    });

    if(code && code !== "0x"){
      walletType = "Smart Contract Wallet";
    }

    document.getElementById("address").innerText =
      address;

    document.getElementById("chainId").innerText =
      chainId;

    document.getElementById("network").innerText =
      networkName;

document.getElementById("balance").innerText =
  Number(balance).toFixed(4) + " ETH";
   
document.getElementById("gas").innerText =
  gasPrice + " GWEI";

document.getElementById("walletAge").innerText =
  walletAge;

document.getElementById("basename").innerText =
  basename;

const walletTypeElement =
      document.getElementById("walletType");

    walletTypeElement.innerText =
      walletType;

    if(walletType.includes("Smart")){
      walletTypeElement.className =
        "value success";
    } else {
      walletTypeElement.className =
        "value warning";
    }

  } catch(err) {

    console.error(err);
    alert("Connection failed");

  }

}

</script>
